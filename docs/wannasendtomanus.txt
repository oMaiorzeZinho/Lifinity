function sortTasks(tasks) {
    if (addon && tasks && tasks.length > 0) {
        try {
            return addon.sortTasks(tasks);
        } catch (error) {
            console.log("⚠️ Erro no sortTasks C, usando fallback JS");
        }
    }
    // Fallback JavaScript
    return [...tasks].sort((a, b) => {
        if (a.is_completed !== b.is_completed) return a.is_completed - b.is_completed;
        const priorityOrder = { high: 1, medium: 2, low: 3 };
        return (priorityOrder[a.priority] || 2) - (priorityOrder[b.priority] || 2);
    });
}

function calculateTaskXP(priority = 'medium', completedOnTime = false, streakDays = 0) {
    if (addon) {
        return addon.calculateTaskXP(priority, completedOnTime, streakDays);
    }
    // Fallback JavaScript
    let xp = 10;
    if (priority === 'high') xp += 5;
    if (priority === 'low') xp -= 3;
    if (completedOnTime) xp += 3;
    if (streakDays >= 7) xp += 5;
    else if (streakDays >= 3) xp += 2;
    return Math.max(1, xp);
}

    let level = 1;
    let xpNeeded = 100;

    while (accumulated + xpNeeded <= totalXP) {
        accumulated += xpNeeded;
        level++;
        xpNeeded = 100 + (level - 1) * 50;

    let accumulated = 0;
    for (let i = 1; i < currentLevel; i++) {
        accumulated += 100 + (i - 1) * 50;
    }
    const xpInCurrentLevel = totalXP - accumulated;
    const xpNeeded = 100 + (currentLevel - 1) * 50;

// FUNÇÃO 1: Ordenar Tarefas
Napi::Array SortTasks(const Napi::CallbackInfo& info) {
    Napi::Env env = info.Env();
    
    if (info.Length() < 1 || !info[0].IsArray()) {
        Napi::TypeError::New(env, "Esperado um array de tarefas").ThrowAsJavaScriptException();
        return Napi::Array::New(env, 0);
    }
    
    Napi::Array jsArray = info[0].As<Napi::Array>();
    uint32_t length = jsArray.Length();
    
    if (length == 0) {
        return Napi::Array::New(env, 0);
    }
    
    std::vector<Task> tasks(length);
    std::vector<Napi::Object> originalTasks(length);
    
    for (uint32_t i = 0; i < length; i++) {
        Napi::Object task = jsArray.Get(i).As<Napi::Object>();
        originalTasks[i] = task;
        
        tasks[i].id = task.Has("id") ? task.Get("id").ToNumber().Int32Value() : (int)i;
        
        std::string priority = task.Has("priority") ? 
            task.Get("priority").ToString().Utf8Value() : "medium";
        
        if (priority == "high") tasks[i].priority = 1;
        else if (priority == "medium") tasks[i].priority = 2;
        else tasks[i].priority = 3;
        
        tasks[i].due_date = 0;
        tasks[i].is_completed = task.Has("is_completed") ? 
            task.Get("is_completed").ToNumber().Int32Value() : 0;
    }
    
    sort_tasks_by_priority_and_date(tasks.data(), length);
    
    Napi::Array result = Napi::Array::New(env, length);
    
    for (uint32_t i = 0; i < length; i++) {
        for (uint32_t j = 0; j < length; j++) {
            int originalId = originalTasks[j].Has("id") ? 
                originalTasks[j].Get("id").ToNumber().Int32Value() : (int)j;
            if (originalId == tasks[i].id) {
                result.Set(i, originalTasks[j]);
                break;
            }
        }
    }
    
    return result;
}

void generate_invite_code(char* buffer, int length) {
    const char charset[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
    int charset_size = 36;
    
    static int initialized = 0;
    if (!initialized) {
        srand((unsigned int)time(NULL));
        initialized = 1;
    }
    
    for (int i = 0; i < length - 1; i++) {
        int key = rand() % charset_size;
        buffer[i] = charset[key];
    }
    buffer[length - 1] = '\0';
}

#include "task_sorter.h"

int compare_tasks(Task* a, Task* b) {
    if (a->is_completed != b->is_completed) {
        return a->is_completed - b->is_completed;
    }
    if (a->priority != b->priority) {
        return a->priority - b->priority;
    }
    if (a->due_date != b->due_date) {
        return (a->due_date < b->due_date) ? -1 : 1;
    }
    return 0;
}

void swap(Task* a, Task* b) {
    Task temp = *a;
    *a = *b;
    *b = temp;
}

int partition(Task* tasks, int low, int high) {
    Task pivot = tasks[high];
    int i = low - 1;
    
    for (int j = low; j < high; j++) {
        if (compare_tasks(&tasks[j], &pivot) <= 0) {
            i++;
            swap(&tasks[i], &tasks[j]);
        }
    }
    swap(&tasks[i + 1], &tasks[high]);
    return i + 1;
}

void quicksort_tasks(Task* tasks, int low, int high) {
    if (low < high) {
        int pi = partition(tasks, low, high);
        quicksort_tasks(tasks, low, pi - 1);
        quicksort_tasks(tasks, pi + 1, high);
    }
}

void sort_tasks_by_priority_and_date(Task* tasks, int count) {
    if (count > 1) {
        quicksort_tasks(tasks, 0, count - 1);
    }
}

int calculate_task_xp(int priority, int completed_on_time, int streak_days) {
    int base_xp = 10;
    
    if (priority == 1) {
        base_xp += 5;
    } else if (priority == 3) {

    }

    }

        base_xp += 5;
    } else if (streak_days >= 3) {
        base_xp += 2;
    }
    
    if (base_xp < 1) {
        base_xp = 1;
    }
    
    return base_xp;
}

    int level = 1;
    int xp_needed = 100;
    int accumulated = 0;
    
    while (accumulated + xp_needed <= total_xp) {
        accumulated += xp_needed;
        level++;
        xp_needed = 100 + (level - 1) * 50;
    }
    
    return level;
}

int xp_for_next_level(int current_level) {
    return 100 + (current_level - 1) * 50;
}

int calculate_level_progress(int total_xp, int current_level) {
    int xp_needed = xp_for_next_level(current_level);
    
    int accumulated = 0;
    int i;
    for (i = 1; i < current_level; i++) {
        accumulated += 100 + (i - 1) * 50;
    }
    
    int xp_in_current_level = total_xp - accumulated;
    int progress = (xp_in_current_level * 100) / xp_needed;
    
    if (progress > 100) progress = 100;
    if (progress < 0) progress = 0;
    
    return progress;
}

export const register = async (req, res) => {
    const { name, email, password } = req.body;

    if (!name || !email || !password) {
        return res.status(400).json({ error: "Preenche todos os campos!" });
    }

    try {
        const [users] = await db.query('SELECT email FROM users WHERE email = ?', [email]);
        if (users.length > 0) {
            return res.status(409).json({ error: "Este email já está registado." });
        }

        const salt = await bcrypt.genSalt(10);
        const hashedPassword = await bcrypt.hash(password, salt);

        const [result] = await db.query(
            'INSERT INTO users (name, email, password) VALUES (?, ?, ?)',
            [name, email, hashedPassword]
        );

        res.status(201).json({ 
            message: "Utilizador criado com sucesso!",
            userId: result.insertId
        });

    } catch (error) {
        res.status(500).json({ error: "Erro no servidor." });
    }
};

// --- LOGIN ---
export const login = async (req, res) => {
    const {email, password} = req.body;

    if(!email || !password){
        return res.status(400).json({ error: "Preenche email e senha!"});
    }

    try{
        //1. Procurar o utilizador pelo email
        const [users] = await db.query('SELECT * FROM users WHERE email = ?', [email]);

        //se não encontrar ninguém
        if(users.length === 0 ){
            return res.status(401).json({error: "Email ou senha incorretos."});
        }

        const user = users[0]; // Pegar o primeiro (e único) utilizador encontrado

        //2. Comparar a senha enviada com a encriptada no BD
        const isMatch = await bcrypt.compare(password, user.password);

        if(!isMatch){
            return res.status(401).json({ error: "Email ou senha incorretos." });
        }

        //3. Gerar o Token
        const token = jwt.sign(
            { id: user.id, name: user.name }, //O que guardamos no token
            JWT_SECRET,                       //A chave Secreta
            { expiresIn: '1h' }               //O token expira em 1 hora
        );

        //4. Enviar tudo de volta
        res.json({
            message: "Login com sucesso!",
            token: token,
            user: { id: user.id, name: user.name, email: user.email }
        });
    }catch(error){
        res.status(500).json({ error: "Erro ao fazer login." });
    }
};

// Extrair data/hora do texto
const extractDateTime = (text) => {
    const now = new Date();
    let dueDate = null;

    if (text.includes('amanhã') || text.includes('amanha')) {
        dueDate = new Date(now);
        dueDate.setDate(dueDate.getDate() + 1);
    } else if (text.includes('depois de amanhã') || text.includes('depois de amanha')) {
        dueDate = new Date(now);
        dueDate.setDate(dueDate.getDate() + 2);
    } else if (text.includes('hoje')) {
        dueDate = new Date(now);
    } else if (text.includes('próxima semana') || text.includes('proxima semana')) {
        dueDate = new Date(now);
        dueDate.setDate(dueDate.getDate() + 7);
    } else if (text.includes('próximo mês') || text.includes('proximo mes')) {
        dueDate = new Date(now);
        dueDate.setMonth(dueDate.getMonth() + 1);
    }

    // Extrair hora (ex: "às 15h", "às 14:30", "15:00")
    const timeMatch = text.match(/(?:às|as|à|a)\s*(\d{1,2})[h:]?(\d{0,2})?/i) || 
                      text.match(/(\d{1,2})[h:](\d{0,2})?/);
    if (timeMatch && dueDate) {
        const hours = parseInt(timeMatch[1]);
        const minutes = timeMatch[2] ? parseInt(timeMatch[2]) : 0;
        if (hours >= 0 && hours <= 23) {
            dueDate.setHours(hours, minutes, 0, 0);
        }
    } else if (dueDate) {
        dueDate.setHours(23, 59, 59, 59);
    }

    return dueDate;
};

// Extrair prioridade do texto
const extractPriority = (text) => {
    if (text.match(/(urgente|muito importante|prioridade alta|alta prioridade|crítico|critico)/i)) {
        return 'high';
    }
    if (text.match(/(baixa prioridade|prioridade baixa|quando puder|sem pressa|calma|tranquilo)/i)) {
        return 'low';
    }
    return 'medium';
};

// Extrair nome do grupo do texto
const extractGroupName = (text) => {
    // Padrões: "para grupo X", "para o grupo X", "grupo X", "para X"
    const groupMatch = text.match(/(?:para\s+(?:o\s+)?grupo\s+|grupo\s+)["']?([^"']+?)["']?(?:\s+|$)/i) ||
                       text.match(/para\s+["']([^"']+)["']/i);
    
    if (groupMatch) {
        return groupMatch[1].trim();
    }
    
    // Verificar "para todos"
    if (text.match(/para\s+todos/i)) {
        return '__ALL__';
    }
    
    return null;
};

// Extrair título da tarefa (limpar palavras-chave)
const extractTaskTitle = (text) => {
    let title = text
        // Remover comandos iniciais
        .replace(/^(criar tarefa|cria tarefa|nova tarefa|adicionar tarefa|adiciona tarefa|lembrar de|lembrar|lembra-me de|lembra-me|tarefa)\s*/i, '')
        // Remover datas
        .replace(/\s*(amanhã|amanha|hoje|depois de amanhã|depois de amanha|próxima semana|proxima semana|próximo mês|proximo mes)\s*/gi, ' ')
        // Remover horas
        .replace(/\s*(?:às|as|à|a)\s*\d{1,2}[h:]?\d{0,2}\s*/gi, ' ')
        .replace(/\s*\d{1,2}[h:]\d{0,2}\s*/gi, ' ')
        // Remover prioridades
        .replace(/\s*(urgente|muito importante|importante|prioridade alta|alta prioridade|baixa prioridade|prioridade baixa|quando puder|sem pressa|crítico|critico)\s*/gi, ' ')
        // Remover referências a grupos
        .replace(/\s*(?:para\s+(?:o\s+)?grupo\s+["']?[^"']+["']?|para\s+["'][^"']+["']|para\s+todos)\s*/gi, ' ')
        // Limpar espaços extras
        .replace(/\s+/g, ' ')
        .trim();

    // Capitalizar primeira letra
    if (title.length > 0) {
        title = title.charAt(0).toUpperCase() + title.slice(1);
    }

    return title || 'Nova tarefa';
};

// Buscar estatísticas do utilizador
const getUserStats = async (userId) => {
    try {
        const [users] = await db.query(
            'SELECT name, xp, level, tasks_completed FROM users WHERE id = ?',
            [userId]
        );

        const [pendingTasks] = await db.query(
            'SELECT COUNT(*) as count FROM tasks WHERE (user_id = ? OR assigned_to = ?) AND is_completed = 0',
            [userId, userId]
        );

        const [completedTasks] = await db.query(
            'SELECT COUNT(*) as count FROM tasks WHERE (user_id = ? OR assigned_to = ?) AND is_completed = 1',
            [userId, userId]
        );

        const [groups] = await db.query(
            'SELECT COUNT(*) as count FROM group_members WHERE user_id = ?',
            [userId]
        );

        return {
            name: users[0]?.name || 'Utilizador',
            xp: users[0]?.xp || 0,
            level: users[0]?.level || 1,
            tasks_completed: users[0]?.tasks_completed || 0,
            pending_tasks: pendingTasks[0]?.count || 0,
            completed_count: completedTasks[0]?.count || 0,
            groups_count: groups[0]?.count || 0
        };
    } catch (error) {
        console.error("Erro ao buscar stats:", error);
        return null;
    }
};

// Buscar grupos do utilizador
const getUserGroups = async (userId) => {
    try {
        const [groups] = await db.query(`
            SELECT g.id, g.name, g.invite_code, gm.role,
                   (SELECT COUNT(*) FROM group_members WHERE group_id = g.id) as member_count
            FROM groups g
            INNER JOIN group_members gm ON g.id = gm.group_id
            WHERE gm.user_id = ?
            ORDER BY g.name
        `, [userId]);
        return groups;
    } catch (error) {
        console.error("Erro ao buscar grupos:", error);
        return [];
    }
};

// Encontrar grupo por nome
const findGroupByName = async (userId, groupName) => {
    try {
        const [groups] = await db.query(`
            SELECT g.id, g.name FROM groups g
            INNER JOIN group_members gm ON g.id = gm.group_id
            WHERE gm.user_id = ? AND LOWER(g.name) LIKE ?
        `, [userId, `%${groupName.toLowerCase()}%`]);
        return groups.length > 0 ? groups[0] : null;
    } catch (error) {
        return null;
    }
};

export const processMessage = async (req, res) => {
    const { message, history = [] } = req.body;
    const userId = req.user.id;

    if (!message || message.trim().length === 0) {
        return res.json({
            response: "🤔 Não percebi. Escreve algo!",
            type: "error"
        });
    }

    const text = message.toLowerCase().trim();

    try {
        // 1️⃣ PRIMEIRO: Tentar comando manual (site)
        const manualResult = await processManualCommand(text, userId);
        
        if (manualResult.handled) {
            console.log("📌 Comando manual:", text);
            return res.json({
                response: manualResult.response,
                type: manualResult.type,
                task: manualResult.task || null
            });
        }

        // 2️⃣ SEGUNDO: Enviar para Gemini AI
        console.log("🤖 Enviando para Gemini:", text);
        const geminiResult = await processWithGemini(message, userId, history);
        
        return res.json({
            response: geminiResult.response,
            type: geminiResult.type
        });

    } catch (error) {
        console.error("Erro no chatbot:", error);
        return res.status(500).json({
            response: "❌ Ocorreu um erro. Tenta novamente.",
            type: "error"
        });
    }
};

            message: "Entraste no grupo com sucesso!",

export const getProfile = async (req, res) => {
    const userId = req.user.id;

    try {
        const [users] = await db.query(
            'SELECT id, name, email, xp, level, tasks_completed, created_at FROM users WHERE id = ?',
            [userId]
        );

        if (users.length === 0) {
            return res.status(404).json({ error: "Utilizador não encontrado." });
        }

        const user = users[0];

        // Usar C para calcular nível e progresso!
        let level, xpProgress, xpNeeded, progressPercent;
        
        if (cAddon && cAddon.isAddonAvailable()) {
            level = cAddon.calculateLevel(user.xp);
            xpNeeded = cAddon.xpForNextLevel(level);
            progressPercent = cAddon.calculateLevelProgress(user.xp, level);
            
            // Calcular XP no nível atual
            let accumulated = 0;
            for (let i = 1; i < level; i++) {
                accumulated += 100 + (i - 1) * 50;
            }
            xpProgress = user.xp - accumulated;
            
            console.log("🔧 Nível calculado por C:", level);
        } else {
            // Fallback JavaScript
            level = user.level;
            xpProgress = user.xp % 100;
            xpNeeded = 100;
            progressPercent = xpProgress;
        }

        // Contar grupos
        const [groups] = await db.query(
            'SELECT COUNT(*) as count FROM group_members WHERE user_id = ?',
            [userId]
        );

        // Estatísticas de tarefas
        const [taskStats] = await db.query(`
            SELECT 
                COUNT(*) as total,
                SUM(CASE WHEN is_completed = 1 THEN 1 ELSE 0 END) as completed,
                SUM(CASE WHEN is_completed = 0 THEN 1 ELSE 0 END) as pending
            FROM tasks 
            WHERE user_id = ? OR assigned_to = ?
        `, [userId, userId]);

        res.json({
            id: user.id,
            name: user.name,
            email: user.email,
            xp: user.xp,
            level: level,
            tasks_completed: user.tasks_completed,
            created_at: user.created_at,
            xp_progress: xpProgress,
            xp_needed: xpNeeded,
            progress_percent: progressPercent,
            groups_count: groups[0].count,
            stats: taskStats[0],
            calculated_by: cAddon && cAddon.isAddonAvailable() ? 'C' : 'JavaScript'
        });

    } catch (error) {
        console.error("Erro ao obter perfil:", error);
        res.status(500).json({ error: "Erro ao carregar perfil." });

export const addXP = async (userId, taskPriority = 'medium') => {
    try {
        // Buscar dados atuais
        const [users] = await db.query('SELECT xp, level FROM users WHERE id = ?', [userId]);
        if (users.length === 0) return null;

        const currentXP = users[0].xp;
        
        // Usar C para calcular XP ganho!
        let xpToAdd;
        if (cAddon && cAddon.isAddonAvailable()) {
            xpToAdd = cAddon.calculateTaskXP(taskPriority, true, 0);
            console.log("🔧 XP calculado por C:", xpToAdd);
        } else {
            // Fallback JavaScript
            xpToAdd = 10;
            if (taskPriority === 'high') xpToAdd += 5;
            if (taskPriority === 'low') xpToAdd -= 3;
        }

        const newXP = currentXP + xpToAdd;
        
        // Usar C para calcular novo nível!
        let newLevel;
        if (cAddon && cAddon.isAddonAvailable()) {
            newLevel = cAddon.calculateLevel(newXP);
        } else {
            newLevel = Math.floor(newXP / 100) + 1;
        }
        
        const leveledUp = newLevel > users[0].level;

        // Atualizar base de dados
        await db.query(
            'UPDATE users SET xp = ?, level = ?, tasks_completed = tasks_completed + 1 WHERE id = ?',
            [newXP, newLevel, userId]
        );

        return { xpGained: xpToAdd, newXP, newLevel, leveledUp };

    } catch (error) {
        console.error("Erro ao adicionar XP:", error);
        return null;
    }
};

export const removeXP = async (userId, taskPriority = 'medium') => {
    try {
        // Buscar dados atuais
        const [users] = await db.query('SELECT xp, level FROM users WHERE id = ?', [userId]);
        if (users.length === 0) return null;

        const currentXP = users[0].xp;
        
        // Usar C para calcular XP a remover!
        let xpToRemove;
        if (cAddon && cAddon.isAddonAvailable()) {
            xpToRemove = cAddon.calculateTaskXP(taskPriority, true, 0);
            console.log("🔧 XP a remover calculado por C:", xpToRemove);
        } else {
            // Fallback JavaScript
            xpToRemove = 10;
            if (taskPriority === 'high') xpToRemove += 5;
            if (taskPriority === 'low') xpToRemove -= 3;
        }

        // Não deixar XP ficar negativo
        const newXP = Math.max(0, currentXP - xpToRemove);
        
        // Usar C para calcular novo nível!
        let newLevel;
        if (cAddon && cAddon.isAddonAvailable()) {
            newLevel = cAddon.calculateLevel(newXP);
        } else {
            newLevel = Math.floor(newXP / 100) + 1;
        }
        
        const leveledDown = newLevel < users[0].level;

        // Atualizar base de dados (diminui tasks_completed também)
        await db.query(
            'UPDATE users SET xp = ?, level = ?, tasks_completed = GREATEST(0, tasks_completed - 1) WHERE id = ?',
            [newXP, newLevel, userId]
        );

        return { xpRemoved: xpToRemove, newXP, newLevel, leveledDown };

    } catch (error) {
        console.error("Erro ao remover XP:", error);
        return null;
    }
};

export const getUserStats = async (req, res) => {
    const userId = req.user.id;

    try {
        // Dados do utilizador
        const [users] = await db.query(
            'SELECT xp, level, tasks_completed, created_at FROM users WHERE id = ?',
            [userId]
        );

        if (users.length === 0) {
            return res.status(404).json({ error: "Utilizador não encontrado." });
        }

        const user = users[0];

        // Total de tarefas (criadas ou atribuídas)
        const [taskCount] = await db.query(`
            SELECT 
                COUNT(*) as total,
                SUM(CASE WHEN is_completed = 1 THEN 1 ELSE 0 END) as completed,
                SUM(CASE WHEN is_completed = 0 THEN 1 ELSE 0 END) as pending
            FROM tasks 
            WHERE user_id = ? OR assigned_to = ?
        `, [userId, userId]);

        // Taxa de conclusão
        const completionRate = taskCount[0].total > 0 
            ? Math.round((taskCount[0].completed / taskCount[0].total) * 100) 
            : 0;

        // Tarefas por prioridade
        const [priorityStats] = await db.query(`
            SELECT 
                priority,
                COUNT(*) as count,
                SUM(CASE WHEN is_completed = 1 THEN 1 ELSE 0 END) as completed
            FROM tasks 
            WHERE user_id = ? OR assigned_to = ?
            GROUP BY priority
        `, [userId, userId]);

        // Tarefas nos últimos 7 dias (para o gráfico de linha)
        const [weeklyStats] = await db.query(`
            SELECT 
                DATE(created_at) as date,
                COUNT(*) as created,
                SUM(CASE WHEN is_completed = 1 THEN 1 ELSE 0 END) as completed
            FROM tasks 
            WHERE (user_id = ? OR assigned_to = ?) 
                AND created_at >= DATE_SUB(NOW(), INTERVAL 7 DAY)
            GROUP BY DATE(created_at)
            ORDER BY date ASC
        `, [userId, userId]);

        // Tarefas por grupo
        const [groupStats] = await db.query(`
            SELECT 
                g.name as group_name,
                COUNT(t.id) as total,
                SUM(CASE WHEN t.is_completed = 1 THEN 1 ELSE 0 END) as completed
            FROM tasks t
            INNER JOIN groups g ON t.group_id = g.id
            WHERE t.group_id IN (SELECT group_id FROM group_members WHERE user_id = ?)
            GROUP BY g.id, g.name
        `, [userId]);

        // Calcular streak (dias consecutivos)
        const [streakData] = await db.query(`
            SELECT DISTINCT DATE(created_at) as date
            FROM tasks
            WHERE (user_id = ? OR assigned_to = ?) AND is_completed = 1
            ORDER BY date DESC
            LIMIT 30
        `, [userId, userId]);

        let currentStreak = 0;
        let checkDate = new Date();
        checkDate.setHours(0, 0, 0, 0);

        for (let i = 0; i < streakData.length; i++) {
            const taskDate = new Date(streakData[i].date);
            taskDate.setHours(0, 0, 0, 0);

            const diffDays = Math.floor((checkDate - taskDate) / (1000 * 60 * 60 * 24));

            if (diffDays === currentStreak) {
                currentStreak++;
            } else {
                break;
            }
        }

        res.json({
            user: {
                xp: user.xp,
                level: user.level,
                tasks_completed: user.tasks_completed,
                member_since: user.created_at
            },
            tasks: {
                total: taskCount[0].total,
                completed: taskCount[0].completed,
                pending: taskCount[0].pending,
                completion_rate: completionRate
            },
            priority_distribution: priorityStats,
            weekly_activity: weeklyStats,
            group_stats: groupStats,
            streak: currentStreak
        });

    } catch (error) {
        console.error("Erro ao obter estatísticas:", error);
        res.status(500).json({ error: "Erro ao carregar estatísticas." });
    }
};

export const updateTaskStatus = async (req, res) => {
    const { id } = req.params;
    const { is_completed } = req.body;
    const userId = req.user.id;

    try {
        const [tasks] = await db.query('SELECT * FROM tasks WHERE id = ?', [id]);

        if (tasks.length === 0) {
            return res.status(404).json({ error: "Tarefa não encontrada." });
        }

        const task = tasks[0];

        // Verificar permissão
        const isCreator = task.user_id === userId;
        const isAssigned = task.assigned_to === userId;
        const hasNoAssignment = task.assigned_to === null;

        if (hasNoAssignment) {
            if (task.group_id) {
                const belongs = await userBelongsToGroup(userId, task.group_id);
                if (!belongs && !isCreator) {
                    return res.status(403).json({ error: "Não tens permissão para concluir esta tarefa." });
                }
            } else if (!isCreator) {
                return res.status(403).json({ error: "Não tens permissão para concluir esta tarefa." });
            }
        } else {
            if (!isAssigned) {
                return res.status(403).json({ 
                    error: "Apenas a pessoa atribuída pode concluir esta tarefa." 
                });
            }
        }

        // Verificar se está a COMPLETAR ou REABRIR
        const wasCompleted = task.is_completed;
        const isCompleting = is_completed && !wasCompleted;
        const isReopening = !is_completed && wasCompleted;

        await db.query('UPDATE tasks SET is_completed = ? WHERE id = ?', [is_completed, id]);

        // DAR XP se está a completar pela primeira vez
        let xpResult = null;
        if (isCompleting) {
            xpResult = await addXP(userId, task.priority);
            console.log("XP adicionado: ", xpResult);
        }
        
        // Remover XP se está a reabrir tarefa
        if(isReopening){
            xpResult = await removeXP(userId, task.priority);
            console.log("XP removido: ", xpResult);
        }

        res.json({ 
            message: isCompleting ? "Tarefa Concluída! " : isReopening ? "Tarefa reaberta." : "Tarefa atualizada!",
            xp: xpResult // Enviar info do XP ganho/perdido
        });

    } catch (error) {
        console.error("Erro ao atualizar:", error);
        res.status(500).json({ error: error.message });
    }
};

import jwt from 'jsonwebtoken';

const JWT_SECRET = 'chave_super_secreta_lifinity_2024'; 

export const verifyToken = (req, res, next) => {
    const token = req.header('Authorization');

    if (!token) return res.status(401).json({ error: "Acesso negado." });

    try {
        const tokenClean = token.replace("Bearer ", "");
        const verified = jwt.verify(tokenClean, JWT_SECRET);
        
        req.user = verified; // <--- ESTA LINHA É CRUCIAL!
        next();
    } catch (error) {
        res.status(400).json({ error: "Token inválido." });
    }
};

import { useState } from 'react';

export default function TaskCard({ task, user, onToggle, onDelete, canComplete }) {
  const [showXP, setShowXP] = useState(false);

  const formatarData = (dataString) => {
    if (!dataString) return "Sem prazo";
    const opcoes = { day: 'numeric', month: 'numeric', year: 'numeric', hour: '2-digit', minute: '2-digit' };
    return new Date(dataString).toLocaleDateString('pt-PT', opcoes);
  };

  const getXPAmount = () => {
    if (task.priority === 'high') return 15;
    if (task.priority === 'low') return 7;
    return 10;
  };

  const handleToggle = () => {
    // Se vai completar (ganhar XP)
    if (!task.is_completed && canComplete) {
      // 1. Mostrar animação IMEDIATAMENTE
      setShowXP(true);
      
      // 2. Esperar 1 segundo antes de atualizar a tarefa
      setTimeout(() => {
        onToggle(task.id, task.is_completed);
      }, 1000);
      
      // 3. Esconder a animação depois de 4 segundos
      setTimeout(() => {
        setShowXP(false);
      }, 4000);
    } else {
      // Se é para reabrir, atualiza imediatamente
      onToggle(task.id, task.is_completed);
    }
  };

  return (
    <div className="relative mb-12">
      {/* Wrapper extra para garantir que o badge não é cortado */}
      <div
        className={`p-5 rounded-lg border shadow flex justify-between items-start transition-all ${
          task.is_completed
            ? 'bg-gray-800/50 border-green-500/50 opacity-60'
            : 'bg-gray-800 border-gray-700 hover:border-blue-500'
        }`}
      >
        {/* Animação de XP - FORA do card, acima */}
        {showXP && (
          <div 
            className="fixed top-1/2 left-1/2 transform -translate-x-1/2 -translate-y-1/2 z-[9999] pointer-events-none"
            style={{ 
              animation: 'xpBounce 0.6s ease-in-out infinite',
            }}
          >
            <div 
              className="bg-gradient-to-r from-yellow-400 via-orange-400 to-yellow-500 text-white font-black text-3xl px-8 py-4 rounded-2xl shadow-2xl border-4 border-yellow-200"
              style={{ 
                boxShadow: '0 0 60px rgba(251, 191, 36, 0.8), 0 0 100px rgba(251, 191, 36, 0.4)',
              }}
            >
              +{getXPAmount()} XP! ⭐✨🎉
            </div>
          </div>
        )}

        {/* Info da tarefa */}
        <div className={`flex-1 ${task.is_completed ? 'line-through text-gray-500' : ''}`}>
          <h3 className="font-bold text-lg text-white flex items-center gap-2 flex-wrap">
            {task.title}
            {task.priority === 'high' && (
              <span className="text-xs bg-red-900 text-red-200 px-2 py-0.5 rounded">Alta</span>
            )}
            {task.priority === 'medium' && (
              <span className="text-xs bg-yellow-900 text-yellow-200 px-2 py-0.5 rounded">Média</span>
            )}
            {task.priority === 'low' && (
              <span className="text-xs bg-green-900 text-green-200 px-2 py-0.5 rounded">Baixa</span>
            )}
            {task.group_name && (
              <span className="text-xs bg-purple-900 text-purple-200 px-2 py-0.5 rounded">
                👥 {task.group_name}
              </span>
            )}
          </h3>
          
          <p className="text-gray-400 text-sm mt-1">{task.description}</p>
          
          <div className="flex gap-4 mt-3 text-xs text-gray-500 flex-wrap items-center">
            <p>📅 {formatarData(task.due_date)}</p>
            {task.created_by_name && <p>Criado por: {task.created_by_name}</p>}
            {task.assigned_to_name && (
              <p className="text-blue-400 font-medium">
                👤 Atribuído a: {task.assigned_to_name}
                {task.assigned_to === user?.id && " (tu)"}
              </p>
            )}
          </div>
        </div>

        {/* Botões */}
        <div className="flex flex-col gap-2 items-center ml-4">
          <button
            onClick={handleToggle}
            disabled={!canComplete}
            className={`w-10 h-10 rounded-full border-2 flex items-center justify-center transition ${
              task.is_completed
                ? 'bg-green-500 border-green-500 text-white'
                : canComplete
                  ? 'border-gray-500 text-gray-500 hover:border-green-400 hover:text-green-400 hover:scale-110'
                  : 'border-gray-700 text-gray-700 cursor-not-allowed'
            }`}
            title={
              task.is_completed 
                ? "Reabrir" 
                : canComplete 
                  ? `Concluir (+${getXPAmount()} XP)` 
                  : `Apenas ${task.assigned_to_name} pode concluir`
            }
          >
            {task.is_completed ? '✓' : ''}
          </button>

          <button
            onClick={() => onDelete(task.id)}
            className="w-10 h-10 rounded-full border-2 border-red-500/50 text-red-500 
                       hover:bg-red-500 hover:text-white hover:border-red-500 
                       flex items-center justify-center transition text-sm hover:scale-110"

        <div className="flex flex-wrap gap-1.5 justify-center">
          {aiButtons.map((btn, i) => (
            <button

              disabled={loading}
              className={`${btn.color} text-white text-xs px-3 py-1.5 rounded-full transition disabled:opacity-50`}
            >
              {btn.label}
            </button>
          ))}
        </div>
      </div>

      {/* Área de Mensagens */}
      <div className="flex-1 bg-gray-800/50 rounded-xl border border-gray-700 p-3 overflow-y-auto mb-3">
        <div className="space-y-3">
          {messages.map((msg) => (
            <div
              key={msg.id}
              className={`flex ${msg.type === 'user' ? 'justify-end' : 'justify-start'}`}

              <div className={`max-w-[90%] p-3 rounded-2xl ${getMessageStyle(msg)}`}>
                {msg.type === 'bot' && (
                  <div className="text-xs text-gray-400 mb-1.5 font-medium">
                    {getTypeLabel(msg.messageType)}
                  </div>
                )}
                <div
                  className="text-sm leading-relaxed"
                  dangerouslySetInnerHTML={{ __html: formatText(msg.text) }}
                />
              </div>
            </div>
          ))}

          {loading && (
            <div className="flex justify-start">
              <div className="bg-gray-700 p-3 rounded-2xl rounded-bl-sm">
                <div className="flex items-center gap-2">

                    <span className="w-2 h-2 bg-cyan-400 rounded-full animate-bounce" style={{ animationDelay: '150ms' }}></span>
                    <span className="w-2 h-2 bg-cyan-400 rounded-full animate-bounce" style={{ animationDelay: '300ms' }}></span>
                  </div>
                  <span className="text-gray-400 text-xs">A processar...</span>
                </div>
              </div>
            </div>
          )}

          <div ref={messagesEndRef} />
        </div>
      </div>

      {/* Input */}
      <form onSubmit={sendMessage} className="flex gap-2">
        <input
          type="text"
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="Escreve um comando ou pergunta..."
          className="flex-1 bg-gray-700 border border-gray-600 rounded-xl p-3 text-white text-sm focus:border-cyan-500 focus:outline-none"
          disabled={loading}
        />
        <button
          type="submit"

          className="bg-cyan-600 hover:bg-cyan-500 disabled:bg-gray-600 text-white px-4 py-3 rounded-xl font-bold transition"
        >
          📤
        </button>
      </form>

      {/* Dica */}
      <p className="text-center text-gray-600 text-xs mt-2">
        💡 Exemplo: "criar tarefa Estudar matemática amanhã às 15h urgente para grupo Família"

}

import { useState, useEffect } from 'react';

  const [groupMembers, setGroupMembers] = useState([]);
  const [selectedGroup, setSelectedGroup] = useState(null);

  const [title, setTitle] = useState('');
  const [description, setDescription] = useState('');
  const [dueDate, setDueDate] = useState('');
  const [priority, setPriority] = useState('medium');
  const [assignedTo, setAssignedTo] = useState('');
  
  // Form Grupo
  const [groupName, setGroupName] = useState('');
  const [inviteCode, setInviteCode] = useState('');
  
  // UI State
  const [showGroupForm, setShowGroupForm] = useState(false);
  const [loading, setLoading] = useState(true);
  const [successMessage, setSuccessMessage] = useState('');

  const token = localStorage.getItem('token');

  // --- FETCH GROUPS ---
  const fetchGroups = async () => {
    try {
      const response = await fetch('http://localhost:5000/groups', {
        headers: { 'Authorization': `Bearer ${token}` }
      });
      if (response.ok) {
        const data = await response.json();
        setGroups(data);
      }
    } catch (error) {
      console.error("Erro ao carregar grupos:", error);
    } finally {
      setLoading(false);
    }
  };

  // --- FETCH GROUP MEMBERS ---
  const fetchGroupMembers = async (groupId) => {
    if (!groupId) {
      setGroupMembers([]);
      return;
    }
    try {
      const response = await fetch(`http://localhost:5000/groups/${groupId}/members`, {
        headers: { 'Authorization': `Bearer ${token}` }
      });
      if (response.ok) {
        const data = await response.json();
        setGroupMembers(data);
      }
    } catch (error) {
      console.error("Erro ao carregar membros:", error);
    }
  };

  useEffect(() => {
    fetchGroups();
  }, []);

  useEffect(() => {
    fetchGroupMembers(selectedGroup);
    setAssignedTo('');
  }, [selectedGroup]);

  // --- CRIAR TAREFA ---
  const handleCreateTask = async (e) => {
    e.preventDefault();
    if (!title) return alert("O título é obrigatório!");

    try {
      const response = await fetch('http://localhost:5000/tasks', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${token}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          title,
          description,
          due_date: dueDate,
          priority,
          group_id: selectedGroup,
          assigned_to: assignedTo || null
        })
      });

      const data = await response.json();
      if (response.ok) {
        // Limpar form
        setTitle('');
        setDescription('');
        setDueDate('');
        setPriority('medium');
        setAssignedTo('');
        
        // Mostrar mensagem de sucesso
        setSuccessMessage('✅ Tarefa criada com sucesso!');
        setTimeout(() => setSuccessMessage(''), 3000);
      } else {
        alert(data.error);
      }
    } catch (error) {
      alert("Erro ao criar tarefa");
    }
  };

  // --- CRIAR GRUPO ---
  const handleCreateGroup = async () => {
    if (!groupName) return alert("Nome do grupo obrigatório!");

    try {
      const response = await fetch('http://localhost:5000/groups', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${token}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ name: groupName })
      });

      const data = await response.json();
      if (response.ok) {
        alert(`Grupo criado! Código de convite: ${data.group.invite_code}`);
        setGroups([data.group, ...groups]);
        setGroupName('');
      } else {
        alert(data.error);
      }
    } catch (error) {
      alert("Erro ao criar grupo");
    }
  };

  // --- ENTRAR NUM GRUPO ---
  const handleJoinGroup = async () => {
    if (!inviteCode) return alert("Código obrigatório!");

    try {
      const response = await fetch('http://localhost:5000/groups/join', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${token}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ inviteCode })
      });

      const data = await response.json();
      if (response.ok) {
        alert(`Entraste no grupo: ${data.group.name}`);
        setGroups([data.group, ...groups]);
        setInviteCode('');
      } else {
        alert(data.error);
      }
    } catch (error) {
      alert("Erro ao entrar no grupo");
    }
  };

  // --- SAIR DO GRUPO ---
  const handleLeaveGroup = async (groupId) => {
    if (!confirm("Tens a certeza que queres sair deste grupo?")) return;

    try {
      const response = await fetch(`http://localhost:5000/groups/${groupId}/leave`, {
        method: 'DELETE',
        headers: { 'Authorization': `Bearer ${token}` }
      });

      if (response.ok) {
        setGroups(groups.filter(g => g.id !== groupId));
        if (selectedGroup === groupId) {
          setSelectedGroup(null);
        }
      } else {
        const data = await response.json();
        alert(data.error);
      }
    } catch (error) {
      alert("Erro ao sair do grupo");
    }
  };

    navigator.clipboard.writeText(text);
    alert("Código copiado!");
  };

    const now = new Date();
    now.setMinutes(now.getMinutes() - now.getTimezoneOffset());
    return now.toISOString().slice(0, 16);
  };

  return (
    <div className="max-w-6xl mx-auto p-4 md:p-8">
      
      {/* Mensagem de Sucesso */}
      {successMessage && (
        <div className="mb-6 bg-green-600/20 border border-green-500 text-green-400 px-4 py-3 rounded-lg text-center animate-pulse">

              >
                {showGroupForm ? '✕ Fechar' : '+ Novo'}
              </button>
            </div>
            
            {/* Form Criar/Entrar Grupo */}
            {showGroupForm && (
              <div className="mb-4 p-3 bg-gray-700/50 rounded-lg space-y-3">
                <div>
                  <input
                    type="text"
                    value={groupName}

      let url = 'http://localhost:5000/tasks';
      if (groupId) url += `?group_id=${groupId}`;

      const response = await fetch(url, {
        headers: { 'Authorization': `Bearer ${token}` }
      });
      if (response.ok) {
        const data = await response.json();
        setTasks(data);
      }
    } catch (error) {
      console.error("Erro:", error);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchGroups();
    fetchTasks();
  }, []);

  useEffect(() => {
    setLoading(true);
    fetchTasks(selectedGroup);
  }, [selectedGroup]);

  const toggleTask = async (taskId, currentStatus) => {
    const newStatus = !currentStatus;
    const previousTasks = [...tasks];
    setTasks(tasks.map(t => t.id === taskId ? { ...t, is_completed: newStatus } : t));

    try {
      const response = await fetch(`http://localhost:5000/tasks/${taskId}`, {
        method: 'PUT',
        headers: {
          'Authorization': `Bearer ${token}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({ is_completed: newStatus })
      });

      if (!response.ok) {
        const data = await response.json();
        alert(data.error);
        setTasks(previousTasks);
      }
    } catch (error) {
      setTasks(previousTasks);
    }
  };

  const deleteTask = async (taskId) => {
    if (!confirm("Tens a certeza que queres eliminar esta tarefa?")) return;

    try {
      const response = await fetch(`http://localhost:5000/tasks/${taskId}`, {
        method: 'DELETE',
        headers: { 'Authorization': `Bearer ${token}` }
      });

      if (response.ok) {
        setTasks(tasks.filter(t => t.id !== taskId));
      } else {
        const data = await response.json();
        alert(data.error);
      }
    } catch (error) {
      alert("Erro ao eliminar tarefa");
    }
  };

  const pendingTasks = tasks.filter(t => !t.is_completed);
  const completedTasks = tasks.filter(t => t.is_completed);

  const canComplete = (task) => {
    if (!task.assigned_to) return true;
    return task.assigned_to === user?.id;
  };

  return (
    <div className="max-w-6xl mx-auto p-4 md:p-8">
      
      {/* Filtro por Grupo */}
      <div className="mb-6">
        <div className="flex flex-wrap gap-2">
          <button
            onClick={() => setSelectedGroup(null)}
            className={`px-4 py-2 rounded-lg transition ${
              selectedGroup === null 
                ? 'bg-blue-600 text-white' 
                : 'bg-gray-700 text-gray-300 hover:bg-gray-600'
            }`}
          >
            📋 Todas
          </button>
          
          {groups.map(group => (
            <button
              key={group.id}
              onClick={() => setSelectedGroup(group.id)}
              className={`px-4 py-2 rounded-lg transition ${
                selectedGroup === group.id 
                  ? 'bg-purple-600 text-white' 
                  : 'bg-gray-700 text-gray-300 hover:bg-gray-600'
              }`}
            >
              👥 {group.name}
            </button>
          ))}
        </div>
      </div>

      {loading ? (
        <div className="text-center py-12">
          <p className="text-gray-400 animate-pulse text-lg">A carregar tarefas...</p>
        </div>
      ) : (
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
          
          {/* COLUNA ESQUERDA - Pendentes */}
          <div>
            <h2 className="text-xl font-bold mb-4 text-blue-400 flex justify-between items-center sticky top-0 bg-gray-900 py-2">
              <span>📋 Por Fazer</span>
              <span className="text-sm bg-blue-600 px-3 py-1 rounded-full">{pendingTasks.length}</span>
            </h2>

            {pendingTasks.length === 0 ? (
              <div className="bg-gray-800/50 border border-dashed border-gray-700 rounded-xl p-8 text-center">
                <p className="text-4xl mb-2">🎉</p>
                <p className="text-gray-500">Nenhuma tarefa pendente!</p>
              </div>
            ) : (
              <div className="space-y-4 pr-2" style={{ maxHeight: '600px', overflowY: 'auto', overflowX: 'visible' }}>
                {pendingTasks.map((task) => (
                  <TaskCard 
                    key={task.id} 
                    task={task} 
                    user={user}
                    onToggle={toggleTask}
                    onDelete={deleteTask}
                    canComplete={canComplete(task)}
                  />
                ))}
              </div>
            )}
          </div>

          {/* COLUNA DIREITA - Concluídas */}
          <div>
            <h2 className="text-xl font-bold mb-4 text-green-400 flex justify-between items-center sticky top-0 bg-gray-900 py-2">
              <span>✅ Concluídas</span>
              <span className="text-sm bg-green-600 px-3 py-1 rounded-full">{completedTasks.length}</span>
            </h2>

            {completedTasks.length === 0 ? (
              <div className="bg-gray-800/50 border border-dashed border-gray-700 rounded-xl p-8 text-center">
                <p className="text-4xl mb-2">📝</p>
                <p className="text-gray-500">Nenhuma tarefa concluída ainda.</p>
              </div>
            ) : (
              <div className="space-y-4 pr-2" style={{ maxHeight: '600px', overflowY: 'auto', overflowX: 'visible' }}>
                {completedTasks.map((task) => (
                  <TaskCard 
                    key={task.id} 
                    task={task} 
                    user={user}
                    onToggle={toggleTask}
                    onDelete={deleteTask}
                    canComplete={canComplete(task)}
                  />
                ))}
              </div>
            )}
          </div>
        </div>
      )}
    </div>
  );
}

lifinity/web/App.jsx -
import { useState, useEffect } from "react";
import { BrowserRouter, Routes, Route, Navigate } from "react-router-dom";

import Login from "./components/Login";
import Register from "./components/Register";
import Navbar from "./components/Navbar";

import HomePage from "./pages/HomePage";
import CreatePage from "./pages/CreatePage";
import TasksPage from "./pages/TasksPage";
import ProfilePage from "./pages/ProfilePage";
import StatsPage from "./pages/StatsPage";
import ChatbotPage from "./pages/ChatbotPage";

function App() {
  const [token, setToken] = useState(localStorage.getItem('token'));
  const [user, setUser] = useState(null);
  const [currentView, setCurrentView] = useState('login');

  useEffect(() => {
    const savedUser = localStorage.getItem('user');
    if (savedUser) {
      setUser(JSON.parse(savedUser));
    }
  }, []);

  const handleLoginSuccess = (newToken, userData) => {
    localStorage.setItem('token', newToken);
    localStorage.setItem('user', JSON.stringify(userData));
    setToken(newToken);
    setUser(userData);
  };

  const handleLogout = () => {
    localStorage.removeItem('token');
    localStorage.removeItem('user');
    setToken(null);
    setUser(null);
    setCurrentView('login');
  };

  if (!token) {
    if (currentView === 'register') {
      return <Register onSwitchToLogin={() => setCurrentView('login')} />;
    }
    return (
      <Login 
        onLoginSuccess={handleLoginSuccess} 
        onSwitchToRegister={() => setCurrentView('register')} 
      />
    );
  }

  return (
    <BrowserRouter>
      <div className="min-h-screen bg-gray-900 text-white">
        <Navbar user={user} onLogout={handleLogout} />
        
        <Routes>
          <Route path="/" element={<Navigate to="/home" replace />} />
          <Route path="/home" element={<HomePage />} />
          <Route path="/create" element={<CreatePage user={user} />} />
          <Route path="/tasks" element={<TasksPage user={user} />} />
          <Route path="/profile" element={<ProfilePage user={user} />} />
          <Route path="/stats" element={<StatsPage user={user} />} />
          <Route path="/chatbot" element={<ChatbotPage user={user} />} />
        </Routes>
      </div>
    </BrowserRouter>
  );
}

export default App;
